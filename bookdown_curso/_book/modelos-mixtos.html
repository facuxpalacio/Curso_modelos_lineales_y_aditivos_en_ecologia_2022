set.seed(101)
tiempo <- seq(1, 20, length = 30)
a <- 20
b <- 5
c <- 0.3
peso <- a - (a - b)*exp(-c*tiempo) + rnorm(n = 30, mean = 0, sd = 0.5)
plot(tiempo, peso)

# Gráfico de residuos
m.crec <- lm(log(peso) ~ log(tiempo))

abline(m.crec, lwd = 2, col = "blue")

plot(log(tiempo), resid(m.crec), xlab = "log(tiempo)", ylab = "Residuos")
